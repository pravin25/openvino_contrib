1) cd JiehongLin-SAM-6D
   cd SAM-6D
   conda env create -f ov_environment_u24.yaml
   conda activate ov_sam6d

2) download nighly build openvino package 
   https://storage.openvinotoolkit.org/repositories/openvino/packages/nightly/
   download following package 
   openvino_toolkit_ubuntu24_2026.3.0.dev20260519_x86_64.tgz
   extract it 
   tar -xvf openvino_toolkit_ubuntu24_2026.3.0.dev20260519_x86_64.tgz
   source openvino_toolkit_ubuntu24_2026.3.0.dev20260519_x86_64/setupvars.sh

3) Install eigen Library
   sudo apt install libeigen3-dev

4) Render templates 
   pip install blenderproc==2.6.1
   cd JiehongLin-SAM-6D/SAM-6D/Render
   # First run downloads Blender 3.3.1 to:
   #   /home/user/blender/blender-3.3.1-linux-x64
   # This requires network access.
   blenderproc run render_custom_templates.py --output_dir ../Data/Example/outputs --cad_path ../Data/Example/obj_000005.ply

   # If the automatic Blender download is blocked, install Blender manually
   # and point BlenderProc at that installation:
   blenderproc run --custom-blender-path /path/to/blender-3.3.1-linux-x64 render_custom_templates.py --output_dir ../Data/Example/outputs --cad_path ../Data/Example/obj_000005.ply

5) cd JiehongLin-SAM-6D/SAM-6D/Instance_Segmentation_Model
   python download_sam.py
   python download_fastsam.py
   python download_dinov2.py
   ##export pytorch to xml  
   python export_ism.py  --ext   --output_dir ./checkpoints/ov_models   --sam_checkpoint_dir ./checkpoints/segment-anything   --dinov2_checkpoint_dir ./checkpoints/dinov2   --fastsam_checkpoint ./checkpoints/FastSAM/FastSAM-x.pt
   ##run ISM
   python infer_ism_ov.py --ov_model_dir ./checkpoints/ov_models --ov_device GPU.0 --ext  --segmentor_model fastsam  --image ../Data/Example/rgb.png --cad ../Data/Example/obj_000005.ply --templates_dir ../Data/Example/outputs/templates  --output_dir ../Data/Example/outputs/sam6d_results  --gt_mask ../Data/Example/mask_visib

6) cd JiehongLin-SAM-6D/SAM-6D/Pose_Estimation_Model/
   python download_sam6d-pem.py 

7) cd JiehongLin-SAM-6D/SAM-6D/Pose_Estimation_Model/model/ov_pointnet2_op
   mkdir build && cd build
   cmake -DCMAKE_BUILD_TYPE=Release -DCMAKE_CXX_FLAGS="-fPIC" -DOpenVINO_DIR=$(python3 -c "from openvino.utils import get_cmake_path; print(get_cmake_path(), end='')") ../ 
   make 

8) ### Install pointnet2
   cd JiehongLin-SAM-6D/SAM-6D/Pose_Estimation_Model/model/pointnet2
   python setup.py install

9) ### run onnx to xml convert script
   cd JiehongLin-SAM-6D/SAM-6D/Pose_Estimation_Model
   python pem_model_convert_cpu.py

10) cd JiehongLin-SAM-6D/SAM-6D/Pose_Estimation_Model
    python run_inference_custom_openvino.py --device GPU --topk_ism_score 1
